SELECT CUSTOMER_NAME FROM CUSTOMER WHERE CUSTOMER_NAME LIKE 'B%';

SELECT COUNT(*) FROM CUSTOMER WHERE BALANCE > CREDIT_LIMIT;

SELECT REP_NUM, SUM(BALANCE) FROM CUSTOMER GROUP BY REP_NUM;

SELECT CUSTOMER.CUSTOMER_NUM, CUSTOMER.CUSTOMER_NAME FROM CUSTOMER
JOIN ORDERS ON CUSTOMER.CUSTOMER_NUM = ORDERS.CUSTOMER_NUM JOIN ORDER_LINE ON ORDERS.ORDER_NUM = ORDER_LINE.ORDER_NUM
WHERE ORDER_LINE.PART_NUM = 'DR93';

SELECT ORDER_NUM, ORDER_DATE FROM ORDERS WHERE CUSTOMER_NUM = '608'

SELECT PART.PART_NUM, PART.DESCRIPTION, PART.ON_HAND FROM PART 
JOIN ORDER_LINE ON PART.PART_NUM = ORDER_LINE.PART_NUM JOIN ORDERS ON ORDERS.ORDER_NUM = ORDER_LINE.ORDER_NUM 
GROUP BY ORDER_LINE.PART_NUM;

CREATE TABLE NONAPPLIANCE
(PART_NUM CHAR(4) PRIMARY KEY,
DESCRIPTION CHAR(15),
ON_HAND DECIMAL(4,0),
CLASS CHAR(2),
PRICE DECIMAL(6,2) );
